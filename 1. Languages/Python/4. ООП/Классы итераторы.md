## ОДНОМЕРНЫЙ ИТЕРАТОР

```python
class Myiter:
    def __init__(self, limit):
        self.num = 0
        self.lim = limit

    def __iter__(self):
        return self

    def __next__(self):
        if self.num >= self.lim:
            raise StopIteration
        self.num += 1
        return self.num
```

## МНОГОМЕРНЫЙ ИТЕРАТОР(2)

```python
class Myiter:
    def __init__(self, x, y):
        self.x = x
        self.y = y
        self.baze = {}

    def __iter__(self):
        return IterOne(self)

class IterOne:
    def __init__(self, obj):
        self.num = 0
        self.limit = obj.y
        self.obj = obj

    def __iter__(self):
        return self

    def __next__(self):
        if self.num >= self.limit:
            raise StopIteration
        self.num += 1
        return IterTwo(self.obj, self.num - 1)

class IterTwo:
    def __init__(self, obj, num1):
        self.num = 0
        self.limit = obj.x
        self.num1 = num1
        self.obj = obj

    def __iter__(self):
        return self

    def __next__(self):
        if self.num >= self.limit:
            raise StopIteration
        self.num += 1
        return self.obj.baze[(self.num1, self.num - 1)]
```
