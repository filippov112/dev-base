```python
from flask import Flask, render_template  # подключаем функцию шаблонизатора render_template
# все шаблоны html документов по умолчанию храняться в папке templates,
# что находится в главном каталоге
# для работы с шаблонами Flask использует модуль Jinja

app = Flask(__name__)

menu = ["a", "b", "c"]  # создаем некоторый список

@app.route("/")
def index():
    return render_template("index.html")  # возвращаем обработчику результат работы функции render,
                                            # со ссылкой на нужный документ

@app.route("/about")
def about():
    return render_template("about.html", title="O сайте")  # передаём значение в аргумент title
    # return render_template("about.html", menu=menu)  # передаём созданный список в шаблон

if __name__ == "__main__":
    app.run(debug=True)
```

## Работа с шаблоном:

```html
{% extends 'base.html' %}  # наследование шаблона по готовому образцу base
```
